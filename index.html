Open the modal on page load

(function($){
    // Cache the selector
    var $modalElement = $('#modalElement');
    // Start the module...
    $modalElement.modal();
    // Or simply start the module by calling a method on it. This will both create the modal (if not created) and open it
    $modalElement.modal('open');

    if($modalElement.modal('is open')){
        alert('Modal is open');
    }
})(mink.$);

Change your settings after runtime

(function($){
    $("#modalElement").modal('setting', 'closeOnEscape', true)
})(mink.$);

(function($){
    $("#modalElement").modal('setting', 'closeOnClick', true)
})(mink.$);

(function($){
    $("#modalElement").modal('internal', 'onClose', function(){
        alert("Modal Closing");
    });
})(mink.$);
